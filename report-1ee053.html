180__isl_give__attribute__((cf_returns_retained)) PWACtx SCEVAffinator::checkForWrapping(const SCEV *Expr,
181 PWACtx PWAC) const {
182 // If the SCEV flags do contain NSW (no signed wrap) then PWA already
183 // represents Expr in modulo semantic (it is not allowed to overflow), thus we
184 // are done. Otherwise, we will compute:
185 // PWA = ((PWA + 2^(n-1)) mod (2 ^ n)) - 2^(n-1)
186 // whereas n is the number of bits of the Expr, hence:
187 // n = bitwidth(ExprType)
188
189 if (IgnoreIntegerWrapping || (getNoWrapFlags(Expr) & SCEV::FlagNSW))
190 return PWAC;
191
192 auto *PWA = PWAC.first;
193 auto *PWAMod = addModuloSemantic(isl_pw_aff_copy(PWA), Expr->getType());
194 auto *NotEqualSet = isl_pw_aff_ne_set(isl_pw_aff_copy(PWA), PWAMod);
195 PWAC.second = isl_set_union(PWAC.second, isl_set_copy(NotEqualSet));
196 PWAC.second = isl_set_coalesce(PWAC.second);
197
198 const DebugLoc &Loc = BB ? BB->getTerminator()->getDebugLoc() : DebugLoc();
199 NotEqualSet = BB ? NotEqualSet : isl_set_params(NotEqualSet);
200 NotEqualSet = isl_set_coalesce(NotEqualSet);
201
202 if (isl_set_is_empty(NotEqualSet))
203 isl_set_free(NotEqualSet);
204 else
205 S->recordAssumption(WRAPPING, NotEqualSet, Loc, AS_RESTRICTION, BB);
206
207 return PWAC;
208}

210__isl_give__attribute__((cf_returns_retained)) isl_pw_aff *
211SCEVAffinator::addModuloSemantic(__isl_take__attribute__((cf_consumed)) isl_pw_aff *PWA,
212 Type *ExprType) const {
213 unsigned Width = TD.getTypeSizeInBits(ExprType);
214 isl_ctx *Ctx = isl_pw_aff_get_ctx(PWA);
215
216 isl_val *ModVal = isl_val_int_from_ui(Ctx, Width);
217 ModVal = isl_val_2exp(ModVal);
218
219 isl_set *Domain = isl_pw_aff_domain(isl_pw_aff_copy(PWA));
220 isl_pw_aff *AddPW = getWidthExpValOnDomain(Width - 1, Domain);
221
222 PWA = isl_pw_aff_add(PWA, isl_pw_aff_copy(AddPW));
223 PWA = isl_pw_aff_mod_val(PWA, ModVal);
224 PWA = isl_pw_aff_sub(PWA, AddPW);
225
226 return PWA;
227}

243__isl_give__attribute__((cf_returns_retained)) PWACtx SCEVAffinator::visit(const SCEV *Expr) {
244
245 auto Key = std::make_pair(Expr, BB);
246 PWACtx PWAC = CachedExpressions[Key];
247 if (PWAC.first)
248 return copyPWACtx(PWAC);
249
250 auto ConstantAndLeftOverPair = extractConstantFactor(Expr, SE);
251 auto *Factor = ConstantAndLeftOverPair.first;
252 Expr = ConstantAndLeftOverPair.second;
253
254 // In case the scev is a valid parameter, we do not further analyze this
255 // expression, but create a new parameter in the isl_pw_aff. This allows us
256 // to treat subexpressions that we cannot translate into an piecewise affine
257 // expression, as constant parameters of the piecewise affine expression.
258 if (isl_id *Id = S->getIdForParam(Expr)) {
259 isl_space *Space = isl_space_set_alloc(Ctx, 1, NumIterators);
260 Space = isl_space_set_dim_id(Space, isl_dim_param, 0, Id);
261
262 isl_set *Domain = isl_set_universe(isl_space_copy(Space));
263 isl_aff *Affine = isl_aff_zero_on_domain(isl_local_space_from_space(Space));
264 Affine = isl_aff_add_coefficient_si(Affine, isl_dim_param, 0, 1);
265
266 PWAC = getPWACtxFromPWA(isl_pw_aff_alloc(Domain, Affine));
267 } else {
268 PWAC = SCEVVisitor<SCEVAffinator, PWACtx>::visit(Expr);
269 PWAC = checkForWrapping(Expr, PWAC);
270 }
271
272 if (!Factor->getType()->isIntegerTy(1))
273 combine(PWAC, visitConstant(Factor), isl_pw_aff_mul);
274
275 // For compile time reasons we need to simplify the PWAC before we cache and
276 // return it.
277 PWAC.first = isl_pw_aff_coalesce(PWAC.first);
278 PWAC = checkForWrapping(Key.first, PWAC);
279
280 CachedExpressions[Key] = copyPWACtx(PWAC);
281 return PWAC;
282}

284__isl_give__attribute__((cf_returns_retained)) PWACtx SCEVAffinator::visitConstant(const SCEVConstant *Expr) {
285 ConstantInt *Value = Expr->getValue();
286 isl_val *v;
287
288 // LLVM does not define if an integer value is interpreted as a signed or
289 // unsigned value. Hence, without further information, it is unknown how
290 // this value needs to be converted to GMP. At the moment, we only support
291 // signed operations. So we just interpret it as signed. Later, there are
292 // two options:
293 //
294 // 1. We always interpret any value as signed and convert the values on
295 // demand.
296 // 2. We pass down the signedness of the calculation and use it to interpret
297 // this constant correctly.
298 v = isl_valFromAPInt(Ctx, Value->getValue(), /* isSigned */ true);
299
300 isl_space *Space = isl_space_set_alloc(Ctx, 0, NumIterators);
301 isl_local_space *ls = isl_local_space_from_space(Space);
302 return getPWACtxFromPWA(isl_pw_aff_from_aff(isl_aff_val_on_domain(ls, v)));
303}

305__isl_give__attribute__((cf_returns_retained)) PWACtx
306SCEVAffinator::visitTruncateExpr(const SCEVTruncateExpr *Expr) {
307 // Truncate operations are basically modulo operations, thus we can
308 // model them that way. However, for large types we assume the operand
309 // to fit in the new type size instead of introducing a modulo with a very
310 // large constant.
311
312 auto *Op = Expr->getOperand();
313 auto OpPWAC = visit(Op);
314
315 unsigned Width = TD.getTypeSizeInBits(Expr->getType());
316 bool Precise = isPreciseTruncate(Width);
317
318 if (Precise) {
319 OpPWAC.first = addModuloSemantic(OpPWAC.first, Expr->getType());
320 return OpPWAC;
321 }
322
323 auto *Dom = isl_pw_aff_domain(isl_pw_aff_copy(OpPWAC.first));
324 auto *ExpPWA = getWidthExpValOnDomain(Width - 1, Dom);
325 auto *GreaterDom =
326 isl_pw_aff_ge_set(isl_pw_aff_copy(OpPWAC.first), isl_pw_aff_copy(ExpPWA));
327 auto *SmallerDom =
328 isl_pw_aff_lt_set(isl_pw_aff_copy(OpPWAC.first), isl_pw_aff_neg(ExpPWA));
329 auto *OutOfBoundsDom = isl_set_union(SmallerDom, GreaterDom);
330 OpPWAC.second = isl_set_union(OpPWAC.second, isl_set_copy(OutOfBoundsDom));
331 S->recordAssumption(UNSIGNED, OutOfBoundsDom, DebugLoc(), AS_RESTRICTION, BB);
332
333 return OpPWAC;
334}

336__isl_give__attribute__((cf_returns_retained)) PWACtx
337SCEVAffinator::visitZeroExtendExpr(const SCEVZeroExtendExpr *Expr) {
338 // A zero-extended value can be interpreted as a piecewise defined signed
339 // value. If the value was non-negative it stays the same, otherwise it
340 // is the sum of the original value and 2^n where n is the bit-width of
341 // the original (or operand) type. Examples:
342 // zext i8 127 to i32 -> { [127] }
343 // zext i8 -1 to i32 -> { [256 + (-1)] } = { [255] }
344 // zext i8 %v to i32 -> [v] -> { [v] | v >= 0; [256 + v] | v < 0 }
345 //
346 // However, LLVM/Scalar Evolution uses zero-extend (potentially lead by a
347 // truncate) to represent some forms of modulo computation. The left-hand side
348 // of the condition in the code below would result in the SCEV
349 // "zext i1 <false, +, true>for.body" which is just another description
350 // of the C expression "i & 1 != 0" or, equivalently, "i % 2 != 0".
351 //
352 // for (i = 0; i < N; i++)
353 // if (i & 1 != 0 /* == i % 2 */)
354 // /* do something */
355 //
356 // If we do not make the modulo explicit but only use the mechanism described
357 // above we will get the very restrictive assumption "N < 3", because for all
358 // values of N >= 3 the SCEVAddRecExpr operand of the zero-extend would wrap.
359 // Alternatively, we can make the modulo in the operand explicit in the
360 // resulting piecewise function and thereby avoid the assumption on N. For the
361 // example this would result in the following piecewise affine function:
362 // { [i0] -> [(1)] : 2*floor((-1 + i0)/2) = -1 + i0;
363 // [i0] -> [(0)] : 2*floor((i0)/2) = i0 }
364 // To this end we can first determine if the (immediate) operand of the
365 // zero-extend can wrap and, in case it might, we will use explicit modulo
366 // semantic to compute the result instead of emitting non-wrapping
367 // assumptions.
368 //
369 // Note that operands with large bit-widths are less likely to be negative
370 // because it would result in a very large access offset or loop bound after
371 // the zero-extend. To this end one can optimistically assume the operand to
372 // be positive and avoid the piecewise definition if the bit-width is bigger
373 // than some threshold (here MaxZextSmallBitWidth).
374 //
375 // We choose to go with a hybrid solution of all modeling techniques described
376 // above. For small bit-widths (up to MaxZextSmallBitWidth) we will model the
377 // wrapping explicitly and use a piecewise defined function. However, if the
378 // bit-width is bigger than MaxZextSmallBitWidth we will employ overflow
379 // assumptions and assume the "former negative" piece will not exist.
380
381 auto *Op = Expr->getOperand();
382 unsigned Width = TD.getTypeSizeInBits(Op->getType());
383
384 bool Precise = isPreciseZeroExtend(Width);
385
386 auto Flags = getNoWrapFlags(Op);
387 auto NoWrapFlags = ScalarEvolution::setFlags(Flags, SCEV::FlagNSW);
388 bool OpCanWrap = Precise && !(Flags & SCEV::FlagNSW);
389 if (OpCanWrap)
390 Op = setNoWrapFlags(SE, Op, NoWrapFlags);
391
392 auto OpPWAC = visit(Op);
393 if (OpCanWrap)
394 OpPWAC.first = addModuloSemantic(OpPWAC.first, Op->getType());
395
396 // If the width is to big we assume the negative part does not occur.
397 if (!Precise) {
398 takeNonNegativeAssumption(OpPWAC);
399 return OpPWAC;
400 }
401
402 // If the width is small build the piece for the non-negative part and
403 // the one for the negative part and unify them.
404 interpretAsUnsigned(OpPWAC, Width);
405 return OpPWAC;
406}

493__isl_give__attribute__((cf_returns_retained)) PWACtx SCEVAffinator::visitUDivExpr(const SCEVUDivExpr *Expr) {
494 // The handling of unsigned division is basically the same as for signed
495 // division, except the interpretation of the operands. As the divisor
496 // has to be constant in both cases we can simply interpret it as an
497 // unsigned value without additional complexity in the representation.
498 // For the dividend we could choose from the different representation
499 // schemes introduced for zero-extend operations but for now we will
500 // simply use an assumption.
501 auto *Dividend = Expr->getLHS();
502 auto *Divisor = Expr->getRHS();
503 assert(isa<SCEVConstant>(Divisor) &&(static_cast<void> (0))
504 "UDiv is no parameter but has a non-constant RHS.")(static_cast<void> (0));
505
506 auto DividendPWAC = visit(Dividend);
507 auto DivisorPWAC = visit(Divisor);
508
509 if (SE.isKnownNegative(Divisor)) {
510 // Interpret negative divisors unsigned. This is a special case of the
511 // piece-wise defined value described for zero-extends as we already know
512 // the actual value of the constant divisor.
513 unsigned Width = TD.getTypeSizeInBits(Expr->getType());
514 auto *DivisorDom = isl_pw_aff_domain(isl_pw_aff_copy(DivisorPWAC.first));
515 auto *WidthExpPWA = getWidthExpValOnDomain(Width, DivisorDom);
516 DivisorPWAC.first = isl_pw_aff_add(DivisorPWAC.first, WidthExpPWA);
517 }
518
519 // TODO: One can represent the dividend as piece-wise function to be more
520 // precise but therefor a heuristic is needed.
521
522 // Assume a non-negative dividend.
523 takeNonNegativeAssumption(DividendPWAC);
524
525 combine(DividendPWAC, DivisorPWAC, isl_pw_aff_div);
526 DividendPWAC.first = isl_pw_aff_floor(DividendPWAC.first);
527
528 return DividendPWAC;
529}

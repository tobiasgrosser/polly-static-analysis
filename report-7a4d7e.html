331/* Construct a function from tagged iteration domains to the corresponding
332 * untagged iteration domains with as range of the wrapped map in the domain
333 * the reference tags that appear in any of the reads, writes or kills.
334 * Store the result in ps->tagger.
335 *
336 * For example, if the statement with iteration space S[i,j]
337 * contains two array references R_1[] and R_2[], then ps->tagger will contain
338 *
339 * { [S[i,j] -> R_1[]] -> S[i,j]; [S[i,j] -> R_2[]] -> S[i,j] }
340 */
341void compute_tagger(struct ppcg_scop *ps)
342{
343 isl_union_map *tagged;
344 isl_union_pw_multi_aff *tagger;
345
346 tagged = isl_union_map_copy(ps->tagged_reads);
347 tagged = isl_union_map_union(tagged,
348 isl_union_map_copy(ps->tagged_may_writes));
349 tagged = isl_union_map_union(tagged,
350 isl_union_map_copy(ps->tagged_must_kills));
351 tagged = isl_union_map_universe(tagged);

359/* Compute the live out accesses, i.e., the writes that are
360 * potentially not killed by any kills or any other writes, and
361 * store them in ps->live_out.
362 *
363 * We compute the "dependence" of any "kill" (an explicit kill
364 * or a must write) on any may write.
365 * The may writes with a "depending" kill are definitely killed.
366 * The remaining may writes can potentially be live out.
367 */
368static void compute_live_out(struct ppcg_scop *ps)
369{
370 isl_union_pw_multi_aff *tagger;
371 isl_schedule *schedule;
372 isl_union_map *kills;
373 isl_union_map *exposed;
374 isl_union_map *covering;
375 isl_union_access_info *access;
376 isl_union_flow *flow;
377
378 tagger = isl_union_pw_multi_aff_copy(ps->tagger);
379 schedule = isl_schedule_copy(ps->schedule);
380 schedule = isl_schedule_pullback_union_pw_multi_aff(schedule, tagger);
381 kills = isl_union_map_union(isl_union_map_copy(ps->tagged_must_writes),
382 isl_union_map_copy(ps->tagged_must_kills));
383 access = isl_union_access_info_from_sink(kills);
384 access = isl_union_access_info_set_may_source(access,
385 isl_union_map_copy(ps->tagged_may_writes));
386 access = isl_union_access_info_set_schedule(access, schedule);
387 flow = isl_union_access_info_compute_flow(access);
388 covering = isl_union_flow_get_may_dependence(flow);
389 isl_union_flow_free(flow);
390 exposed = isl_union_map_copy(ps->tagged_may_writes);
391 exposed = isl_union_map_subtract_domain(exposed,
392 isl_union_map_domain(covering));
393 ps->live_out = project_out_tags(exposed);
394}

396/* Compute the tagged flow dependences and the live_in accesses and store
397 * the results in ps->tagged_dep_flow and ps->live_in.
398 *
399 * We allow both the must writes and the must kills to serve as
400 * definite sources such that a subsequent read would not depend
401 * on any earlier write. The resulting flow dependences with
402 * a must kill as source reflect possibly uninitialized reads.
403 * No dependences need to be introduced to protect such reads
404 * (other than those imposed by potential flows from may writes
405 * that follow the kill). We therefore remove those flow dependences.
406 * This is also useful for the dead code elimination, which assumes
407 * the flow sources are non-kill instances.
408 */
409static void compute_tagged_flow_dep_only(struct ppcg_scop *ps)
410{
411 isl_union_pw_multi_aff *tagger;
412 isl_schedule *schedule;
413 isl_union_map *live_in;
414 isl_union_access_info *access;
415 isl_union_flow *flow;
416 isl_union_map *must_source;
417 isl_union_map *kills;
418 isl_union_map *tagged_flow;
419
420 tagger = isl_union_pw_multi_aff_copy(ps->tagger);
421 schedule = isl_schedule_copy(ps->schedule);
422 schedule = isl_schedule_pullback_union_pw_multi_aff(schedule, tagger);
423 kills = isl_union_map_copy(ps->tagged_must_kills);
424 must_source = isl_union_map_copy(ps->tagged_must_writes);
425 must_source = isl_union_map_union(must_source,
426 isl_union_map_copy(kills));
427 access = isl_union_access_info_from_sink(
428 isl_union_map_copy(ps->tagged_reads));
429 access = isl_union_access_info_set_must_source(access, must_source);
430 access = isl_union_access_info_set_may_source(access,
431 isl_union_map_copy(ps->tagged_may_writes));
432 access = isl_union_access_info_set_schedule(access, schedule);
433 flow = isl_union_access_info_compute_flow(access);
434 tagged_flow = isl_union_flow_get_may_dependence(flow);
435 tagged_flow = isl_union_map_subtract_domain(tagged_flow,
436 isl_union_map_domain(kills));
437 ps->tagged_dep_flow = tagged_flow;
438 live_in = isl_union_flow_get_may_no_source(flow);
439 ps->live_in = project_out_tags(live_in);
440 isl_union_flow_free(flow);
441}

466/* Compute the order dependences that prevent the potential live ranges
467 * from overlapping.
468 *
469 * In particular, construct a union of relations
470 *
471 * [R[...] -> R_1[]] -> [W[...] -> R_2[]]
472 *
473 * where [R[...] -> R_1[]] is the range of one or more live ranges
474 * (i.e., a read) and [W[...] -> R_2[]] is the domain of one or more
475 * live ranges (i.e., a write). Moreover, the read and the write
476 * access the same memory element and the read occurs before the write
477 * in the original schedule.
478 * The scheduler allows some of these dependences to be violated, provided
479 * the adjacent live ranges are all local (i.e., their domain and range
480 * are mapped to the same point by the current schedule band).
481 *
482 * Note that if a live range is not local, then we need to make
483 * sure it does not overlap with _any_ other live range, and not
484 * just with the "previous" and/or the "next" live range.
485 * We therefore add order dependences between reads and
486 * _any_ later potential write.
487 *
488 * We also need to be careful about writes without a corresponding read.
489 * They are already prevented from moving past non-local preceding
490 * intervals, but we also need to prevent them from moving past non-local
491 * following intervals. We therefore also add order dependences from
492 * potential writes that do not appear in any intervals
493 * to all later potential writes.
494 * Note that dead code elimination should have removed most of these
495 * dead writes, but the dead code elimination may not remove all dead writes,
496 * so we need to consider them to be safe.
497 *
498 * The order dependences are computed by computing the "dataflow"
499 * from the above unmatched writes and the reads to the may writes.
500 * The unmatched writes and the reads are treated as may sources
501 * such that they would not kill order dependences from earlier
502 * such writes and reads.
503 */
504static void compute_order_dependences(struct ppcg_scop *ps)
505{
506 isl_union_map *reads;
507 isl_union_map *shared_access;
508 isl_union_set *matched;
509 isl_union_map *unmatched;
510 isl_union_pw_multi_aff *tagger;
511 isl_schedule *schedule;
512 isl_union_access_info *access;
513 isl_union_flow *flow;
514
515 tagger = isl_union_pw_multi_aff_copy(ps->tagger);
516 schedule = isl_schedule_copy(ps->schedule);
517 schedule = isl_schedule_pullback_union_pw_multi_aff(schedule, tagger);
518 reads = isl_union_map_copy(ps->tagged_reads);
519 matched = isl_union_map_domain(isl_union_map_copy(ps->tagged_dep_flow));
520 unmatched = isl_union_map_copy(ps->tagged_may_writes);
521 unmatched = isl_union_map_subtract_domain(unmatched, matched);
522 reads = isl_union_map_union(reads, unmatched);
523 access = isl_union_access_info_from_sink(
524 isl_union_map_copy(ps->tagged_may_writes));
525 access = isl_union_access_info_set_may_source(access, reads);
526 access = isl_union_access_info_set_schedule(access, schedule);
527 flow = isl_union_access_info_compute_flow(access);
528 shared_access = isl_union_flow_get_may_dependence(flow);
529 isl_union_flow_free(flow);
530
531 ps->tagged_dep_order = isl_union_map_copy(shared_access);
532 ps->dep_order = isl_union_map_factor_domain(shared_access);
533}

535/* Compute those validity dependences of the program represented by "scop"
536 * that should be unconditionally enforced even when live-range reordering
537 * is used.
538 *
539 * In particular, compute the external false dependences
540 * as well as order dependences between sources with the same sink.
541 * The anti-dependences are already taken care of by the order dependences.
542 * The external false dependences are only used to ensure that live-in and
543 * live-out data is not overwritten by any writes inside the scop.
544 * The independences are removed from the external false dependences,
545 * but not from the order dependences between sources with the same sink.
546 *
547 * In particular, the reads from live-in data need to precede any
548 * later write to the same memory element.
549 * As to live-out data, the last writes need to remain the last writes.
550 * That is, any earlier write in the original schedule needs to precede
551 * the last write to the same memory element in the computed schedule.
552 * The possible last writes have been computed by compute_live_out.
553 * They may include kills, but if the last access is a kill,
554 * then the corresponding dependences will effectively be ignored
555 * since we do not schedule any kill statements.
556 *
557 * Note that the set of live-in and live-out accesses may be
558 * an overapproximation. There may therefore be potential writes
559 * before a live-in access and after a live-out access.
560 *
561 * In the presence of may-writes, there may be multiple live-ranges
562 * with the same sink, accessing the same memory element.
563 * The sources of these live-ranges need to be executed
564 * in the same relative order as in the original program
565 * since we do not know which of the may-writes will actually
566 * perform a write. Consider all sources that share a sink and
567 * that may write to the same memory element and compute
568 * the order dependences among them.
569 */
570static void compute_forced_dependences(struct ppcg_scop *ps)
571{
572 isl_union_map *shared_access;
573 isl_union_map *exposed;
574 isl_union_map *live_in;
575 isl_union_map *sink_access;
576 isl_union_map *shared_sink;
577 isl_union_access_info *access;
578 isl_union_flow *flow;
579 isl_schedule *schedule;
580
581 exposed = isl_union_map_copy(ps->live_out);
582 schedule = isl_schedule_copy(ps->schedule);
583 access = isl_union_access_info_from_sink(exposed);
584 access = isl_union_access_info_set_may_source(access,
585 isl_union_map_copy(ps->may_writes));
586 access = isl_union_access_info_set_schedule(access, schedule);
587 flow = isl_union_access_info_compute_flow(access);
588 shared_access = isl_union_flow_get_may_dependence(flow);
589 isl_union_flow_free(flow);
590 ps->dep_forced = shared_access;
591
592 schedule = isl_schedule_copy(ps->schedule);
593 access = isl_union_access_info_from_sink(
594 isl_union_map_copy(ps->may_writes));
595 access = isl_union_access_info_set_may_source(access,
596 isl_union_map_copy(ps->live_in));
597 access = isl_union_access_info_set_schedule(access, schedule);
598 flow = isl_union_access_info_compute_flow(access);
599 live_in = isl_union_flow_get_may_dependence(flow);
600 isl_union_flow_free(flow);
601
602 ps->dep_forced = isl_union_map_union(ps->dep_forced, live_in);
603 ps->dep_forced = isl_union_map_subtract(ps->dep_forced,
604 isl_union_map_copy(ps->independence));
605
606 schedule = isl_schedule_copy(ps->schedule);
607 sink_access = isl_union_map_copy(ps->tagged_dep_flow);
608 sink_access = isl_union_map_range_product(sink_access,
609 isl_union_map_copy(ps->tagged_may_writes));
610 sink_access = isl_union_map_domain_factor_domain(sink_access);
611 access = isl_union_access_info_from_sink(
612 isl_union_map_copy(sink_access));
613 access = isl_union_access_info_set_may_source(access, sink_access);
614 access = isl_union_access_info_set_schedule(access, schedule);
615 flow = isl_union_access_info_compute_flow(access);
616 shared_sink = isl_union_flow_get_may_dependence(flow);
617 isl_union_flow_free(flow);
618 ps->dep_forced = isl_union_map_union(ps->dep_forced, shared_sink);
619}

621/* Remove independence from the tagged flow dependences.
622 * Since the user has guaranteed that source and sink of an independence
623 * can be executed in any order, there cannot be a flow dependence
624 * between them, so they can be removed from the set of flow dependences.
625 * However, if the source of such a flow dependence is a must write,
626 * then it may have killed other potential sources, which would have
627 * to be recovered if we were to remove those flow dependences.
628 * We therefore keep the flow dependences that originate in a must write,
629 * even if it corresponds to a known independence.
630 */
631static void remove_independences_from_tagged_flow(struct ppcg_scop *ps)
632{
633 isl_union_map *tf;
634 isl_union_set *indep;
635 isl_union_set *mw;
636
637 tf = isl_union_map_copy(ps->tagged_dep_flow);
638 tf = isl_union_map_zip(tf);
639 indep = isl_union_map_wrap(isl_union_map_copy(ps->independence));
640 tf = isl_union_map_intersect_domain(tf, indep);
641 tf = isl_union_map_zip(tf);
642 mw = isl_union_map_domain(isl_union_map_copy(ps->tagged_must_writes));
643 tf = isl_union_map_subtract_domain(tf, mw);
644 ps->tagged_dep_flow = isl_union_map_subtract(ps->tagged_dep_flow, tf);
645}

647/* Compute the dependences of the program represented by "scop"
648 * in case live range reordering is allowed.
649 *
650 * We compute the actual live ranges and the corresponding order
651 * false dependences.
652 *
653 * The independences are removed from the flow dependences
654 * (provided the source is not a must-write) as well as
655 * from the external false dependences (by compute_forced_dependences).
656 */
657static void compute_live_range_reordering_dependences(struct ppcg_scop *ps)
658{
659 compute_tagged_flow_dep_only(ps);
660 remove_independences_from_tagged_flow(ps);
661 derive_flow_dep_from_tagged_flow_dep(ps);
662 compute_order_dependences(ps);
663 compute_forced_dependences(ps);
664}

666/* Compute the potential flow dependences and the potential live in
667 * accesses.
668 */
669static void compute_flow_dep(struct ppcg_scop *ps)
670{
671 isl_union_access_info *access;
672 isl_union_flow *flow;
673
674 access = isl_union_access_info_from_sink(isl_union_map_copy(ps->reads));
675 access = isl_union_access_info_set_must_source(access,
676 isl_union_map_copy(ps->must_writes));
677 access = isl_union_access_info_set_may_source(access,
678 isl_union_map_copy(ps->may_writes));
679 access = isl_union_access_info_set_schedule(access,
680 isl_schedule_copy(ps->schedule));
681 flow = isl_union_access_info_compute_flow(access);
682
683 ps->dep_flow = isl_union_flow_get_may_dependence(flow);
684 ps->live_in = isl_union_flow_get_may_no_source(flow);
685 isl_union_flow_free(flow);
686}

688/* Compute the dependences of the program represented by "scop".
689 * Store the computed potential flow dependences
690 * in scop->dep_flow and the reads with potentially no corresponding writes in
691 * scop->live_in.
692 * Store the potential live out accesses in scop->live_out.
693 * Store the potential false (anti and output) dependences in scop->dep_false.
694 *
695 * If live range reordering is allowed, then we compute a separate
696 * set of order dependences and a set of external false dependences
697 * in compute_live_range_reordering_dependences.
698 */
699void compute_dependences(struct ppcg_scop *scop)
700{
701 isl_union_map *may_source;
702 isl_union_access_info *access;
703 isl_union_flow *flow;
704
705 if (!scop)
706 return;
707
708 compute_live_out(scop);
709
710 if (scop->options->live_range_reordering)
711 compute_live_range_reordering_dependences(scop);
712 else if (scop->options->target != PPCG_TARGET_C0)
713 compute_tagged_flow_dep(scop);
714 else
715 compute_flow_dep(scop);
716
717 may_source = isl_union_map_union(isl_union_map_copy(scop->may_writes),
718 isl_union_map_copy(scop->reads));
719 access = isl_union_access_info_from_sink(
720 isl_union_map_copy(scop->may_writes));
721 access = isl_union_access_info_set_must_source(access,
722 isl_union_map_copy(scop->must_writes));
723 access = isl_union_access_info_set_may_source(access, may_source);
724 access = isl_union_access_info_set_schedule(access,
725 isl_schedule_copy(scop->schedule));
726 flow = isl_union_access_info_compute_flow(access);
727
728 scop->dep_false = isl_union_flow_get_may_dependence(flow);
729 scop->dep_false = isl_union_map_coalesce(scop->dep_false);
730 isl_union_flow_free(flow);
731}

733/* Eliminate dead code from ps->domain.
734 *
735 * In particular, intersect both ps->domain and the domain of
736 * ps->schedule with the (parts of) iteration
737 * domains that are needed to produce the output or for statement
738 * iterations that call functions.
739 * Also intersect the range of the dataflow dependences with
740 * this domain such that the removed instances will no longer
741 * be considered as targets of dataflow.
742 *
743 * We start with the iteration domains that call functions
744 * and the set of iterations that last write to an array
745 * (except those that are later killed).
746 *
747 * Then we add those statement iterations that produce
748 * something needed by the "live" statements iterations.
749 * We keep doing this until no more statement iterations can be added.
750 * To ensure that the procedure terminates, we compute the affine
751 * hull of the live iterations (bounded to the original iteration
752 * domains) each time we have added extra iterations.
753 */
754static void eliminate_dead_code(struct ppcg_scop *ps)
755{
756 isl_union_set *live;
757 isl_union_map *dep;
758 isl_union_pw_multi_aff *tagger;
759
760 live = isl_union_map_domain(isl_union_map_copy(ps->live_out));
761 if (!isl_union_set_is_empty(ps->call)) {
762 live = isl_union_set_union(live, isl_union_set_copy(ps->call));
763 live = isl_union_set_coalesce(live);
764 }
765
766 dep = isl_union_map_copy(ps->dep_flow);
767 dep = isl_union_map_reverse(dep);
768
769 for (;;) {
770 isl_union_set *extra;
771
772 extra = isl_union_set_apply(isl_union_set_copy(live),
773 isl_union_map_copy(dep));
774 if (isl_union_set_is_subset(extra, live)) {
775 isl_union_set_free(extra);
776 break;
777 }
778
779 live = isl_union_set_union(live, extra);
780 live = isl_union_set_affine_hull(live);
781 live = isl_union_set_intersect(live,
782 isl_union_set_copy(ps->domain));
783 }
784
785 isl_union_map_free(dep);
786
787 ps->domain = isl_union_set_intersect(ps->domain,
788 isl_union_set_copy(live));
789 ps->schedule = isl_schedule_intersect_domain(ps->schedule,
790 isl_union_set_copy(live));
791 ps->dep_flow = isl_union_map_intersect_range(ps->dep_flow,
792 isl_union_set_copy(live));
793 tagger = isl_union_pw_multi_aff_copy(ps->tagger);
794 live = isl_union_set_preimage_union_pw_multi_aff(live, tagger);
795 ps->tagged_dep_flow = isl_union_map_intersect_range(ps->tagged_dep_flow,
796 live);
797}

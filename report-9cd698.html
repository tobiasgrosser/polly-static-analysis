197/// Compute the privatization dependences for a given dependency @p Map
198///
199/// Privatization dependences are widened original dependences which originate
200/// or end in a reduction access. To compute them we apply the transitive close
201/// of the reduction dependences (which maps each iteration of a reduction
202/// statement to all following ones) on the RAW/WAR/WAW dependences. The
203/// dependences which start or end at a reduction statement will be extended to
204/// depend on all following reduction statement iterations as well.
205/// Note: "Following" here means according to the reduction dependences.
206///
207/// For the input:
208///
209/// S0: *sum = 0;
210/// for (int i = 0; i < 1024; i++)
211/// S1: *sum += i;
212/// S2: *sum = *sum * 3;
213///
214/// we have the following dependences before we add privatization dependences:
215///
216/// RAW:
217/// { S0[] -> S1[0]; S1[1023] -> S2[] }
218/// WAR:
219/// { }
220/// WAW:
221/// { S0[] -> S1[0]; S1[1024] -> S2[] }
222/// RED:
223/// { S1[i0] -> S1[1 + i0] : i0 >= 0 and i0 <= 1022 }
224///
225/// and afterwards:
226///
227/// RAW:
228/// { S0[] -> S1[i0] : i0 >= 0 and i0 <= 1023;
229/// S1[i0] -> S2[] : i0 >= 0 and i0 <= 1023}
230/// WAR:
231/// { }
232/// WAW:
233/// { S0[] -> S1[i0] : i0 >= 0 and i0 <= 1023;
234/// S1[i0] -> S2[] : i0 >= 0 and i0 <= 1023}
235/// RED:
236/// { S1[i0] -> S1[1 + i0] : i0 >= 0 and i0 <= 1022 }
237///
238/// Note: This function also computes the (reverse) transitive closure of the
239/// reduction dependences.
240void Dependences::addPrivatizationDependences() {
241 isl_union_map *PrivRAW, *PrivWAW, *PrivWAR;
242
243 // The transitive closure might be over approximated, thus could lead to
244 // dependency cycles in the privatization dependences. To make sure this
245 // will not happen we remove all negative dependences after we computed
246 // the transitive closure.
247 TC_RED = isl_union_map_transitive_closure(isl_union_map_copy(RED), nullptr);
248
249 // FIXME: Apply the current schedule instead of assuming the identity schedule
250 // here. The current approach is only valid as long as we compute the
251 // dependences only with the initial (identity schedule). Any other
252 // schedule could change "the direction of the backward dependences" we
253 // want to eliminate here.
254 isl_union_set *UDeltas = isl_union_map_deltas(isl_union_map_copy(TC_RED));
255 isl_union_set *Universe = isl_union_set_universe(isl_union_set_copy(UDeltas));
256 isl_union_set *Zero = isl_union_set_empty(isl_union_set_get_space(Universe));
257 isl_union_set_foreach_set(Universe, fixSetToZero, &Zero);
258 isl_union_map *NonPositive = isl_union_set_lex_le_union_set(UDeltas, Zero);
259
260 TC_RED = isl_union_map_subtract(TC_RED, NonPositive);
261
262 TC_RED = isl_union_map_union(
263 TC_RED, isl_union_map_reverse(isl_union_map_copy(TC_RED)));
264 TC_RED = isl_union_map_coalesce(TC_RED);
265
266 isl_union_map **Maps[] = {&RAW, &WAW, &WAR};
267 isl_union_map **PrivMaps[] = {&PrivRAW, &PrivWAW, &PrivWAR};
268 for (unsigned u = 0; u < 3; u++) {
269 isl_union_map **Map = Maps[u], **PrivMap = PrivMaps[u];
270
271 *PrivMap = isl_union_map_apply_range(isl_union_map_copy(*Map),
272 isl_union_map_copy(TC_RED));
273 *PrivMap = isl_union_map_union(
274 *PrivMap, isl_union_map_apply_range(isl_union_map_copy(TC_RED),
275 isl_union_map_copy(*Map)));
276
277 *Map = isl_union_map_union(*Map, *PrivMap);
278 }
279
280 isl_union_set_free(Universe);
281}

290static __isl_give__attribute__((cf_returns_retained)) isl_union_map *
291addZeroPaddingToSchedule(__isl_take__attribute__((cf_consumed)) isl_union_map *Schedule) {
292 unsigned int MaxScheduleDim = 0;
293
294 isl_union_map_foreach_map(Schedule, getMaxScheduleDim, &MaxScheduleDim);
295
296 auto ExtensionMap = isl_union_map_empty(isl_union_map_get_space(Schedule));
297 for (unsigned int i = 0; i <= MaxScheduleDim; i++) {
298 auto *Map = isl_map_identity(
299 isl_space_alloc(isl_union_map_get_ctx(Schedule), 0, i, i));
300 Map = isl_map_add_dims(Map, isl_dim_out, MaxScheduleDim - i);
301 for (unsigned int j = 0; j < MaxScheduleDim - i; j++)
302 Map = isl_map_fix_si(Map, isl_dim_out, i + j, 0);
303
304 ExtensionMap = isl_union_map_add_map(ExtensionMap, Map);
305 }
306 Schedule = isl_union_map_apply_range(Schedule, ExtensionMap);
307
308 return Schedule;
309}

593bool Dependences::isValidSchedule(Scop &S,
594 StatementToIslMapTy *NewSchedule) const {
595 if (LegalityCheckDisabled)
596 return true;
597
598 isl_union_map *Dependences = getDependences(TYPE_RAW | TYPE_WAW | TYPE_WAR);
599 isl_space *Space = S.getParamSpace();
600 isl_union_map *Schedule = isl_union_map_empty(Space);
601
602 isl_space *ScheduleSpace = nullptr;
603
604 for (ScopStmt &Stmt : S) {
605 isl_map *StmtScat;
606
607 if (NewSchedule->find(&Stmt) == NewSchedule->end())
608 StmtScat = Stmt.getSchedule();
609 else
610 StmtScat = isl_map_copy((*NewSchedule)[&Stmt]);
611
612 if (!ScheduleSpace)
613 ScheduleSpace = isl_space_range(isl_map_get_space(StmtScat));
614
615 Schedule = isl_union_map_add_map(Schedule, StmtScat);
616 }
617
618 Dependences =
619 isl_union_map_apply_domain(Dependences, isl_union_map_copy(Schedule));
620 Dependences = isl_union_map_apply_range(Dependences, Schedule);
621
622 isl_set *Zero = isl_set_universe(isl_space_copy(ScheduleSpace));
623 for (unsigned i = 0; i < isl_set_dim(Zero, isl_dim_set); i++)
624 Zero = isl_set_fix_si(Zero, isl_dim_set, i, 0);
625
626 isl_union_set *UDeltas = isl_union_map_deltas(Dependences);
627 isl_set *Deltas = isl_union_set_extract_set(UDeltas, ScheduleSpace);
628 isl_union_set_free(UDeltas);
629
630 isl_map *NonPositive = isl_set_lex_le_set(Deltas, Zero);
631 bool IsValid = isl_map_is_empty(NonPositive);
632 isl_map_free(NonPositive);
633
634 return IsValid;
635}
636
637// Check if the current scheduling dimension is parallel.
638//
639// We check for parallelism by verifying that the loop does not carry any
640// dependences.
641//
642// Parallelism test: if the distance is zero in all outer dimensions, then it
643// has to be zero in the current dimension as well.
644//
645// Implementation: first, translate dependences into time space, then force
646// outer dimensions to be equal. If the distance is zero in the current
647// dimension, then the loop is parallel. The distance is zero in the current
648// dimension if it is a subset of a map with equal values for the current
649// dimension.
650bool Dependences::isParallel(isl_union_map *Schedule, isl_union_map *Deps,
651 isl_pw_aff **MinDistancePtr) const {
652 isl_set *Deltas, *Distance;
653 isl_map *ScheduleDeps;
654 unsigned Dimension;
655 bool IsParallel;
656
657 Deps = isl_union_map_apply_range(Deps, isl_union_map_copy(Schedule));
658 Deps = isl_union_map_apply_domain(Deps, isl_union_map_copy(Schedule));
659
660 if (isl_union_map_is_empty(Deps)) {
661 isl_union_map_free(Deps);
662 return true;
663 }
664
665 ScheduleDeps = isl_map_from_union_map(Deps);
666 Dimension = isl_map_dim(ScheduleDeps, isl_dim_out) - 1;
667
668 for (unsigned i = 0; i < Dimension; i++)
669 ScheduleDeps = isl_map_equate(ScheduleDeps, isl_dim_out, i, isl_dim_in, i);
670
671 Deltas = isl_map_deltas(ScheduleDeps);
672 Distance = isl_set_universe(isl_set_get_space(Deltas));
673
674 // [0, ..., 0, +] - All zeros and last dimension larger than zero
675 for (unsigned i = 0; i < Dimension; i++)
676 Distance = isl_set_fix_si(Distance, isl_dim_set, i, 0);
677
678 Distance = isl_set_lower_bound_si(Distance, isl_dim_set, Dimension, 1);
679 Distance = isl_set_intersect(Distance, Deltas);
680
681 IsParallel = isl_set_is_empty(Distance);
682 if (IsParallel || !MinDistancePtr) {
683 isl_set_free(Distance);
684 return IsParallel;
685 }
686
687 Distance = isl_set_project_out(Distance, isl_dim_set, 0, Dimension);
688 Distance = isl_set_coalesce(Distance);
689
690 // This last step will compute a expression for the minimal value in the
691 // distance polyhedron Distance with regards to the first (outer most)
692 // dimension.
693 *MinDistancePtr = isl_pw_aff_coalesce(isl_set_dim_min(Distance, 0));
694
695 return false;
696}

735 assert(hasValidDependences() && "No valid dependences available")(static_cast<void> (0));

742 if (Kinds & TYPE_WAR)

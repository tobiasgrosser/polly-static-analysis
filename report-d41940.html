348/* Given part of the schedule of a band, construct the corresponding
349 * schedule for the tile loops based on the tile sizes in data->sizes
350 * and add the result to data->tiled.
351 *
352 * If data->scale is set, then dimension i of the schedule will be
353 * of the form
354 *
355 * m_i * floor(s_i(x) / m_i)
356 *
357 * where s_i(x) refers to the original schedule and m_i is the tile size.
358 * If data->scale is not set, then dimension i of the schedule will be
359 * of the form
360 *
361 * floor(s_i(x) / m_i)
362 *
363 */
364static isl_stat multi_aff_tile(__isl_take__attribute__((cf_consumed)) isl_set *set,
365 __isl_take__attribute__((cf_consumed)) isl_multi_aff *ma, void *user)
366{
367 struct isl_band_tile_data *data = user;
368 isl_pw_multi_aff *pma;
369 int i, n;
370 isl_val *v;
371
372 n = isl_multi_aff_dim(ma, isl_dim_out);
373
374 for (i = 0; i < n; ++i) {
375 isl_aff *aff;
376
377 aff = isl_multi_aff_get_aff(ma, i);
378 v = isl_multi_val_get_val(data->sizes, i);
379
380 aff = isl_aff_scale_down_val(aff, isl_val_copy(v));
381 aff = isl_aff_floor(aff);
382 if (data->scale)
383 aff = isl_aff_scale_val(aff, isl_val_copy(v));
384 isl_val_free(v);
385
386 ma = isl_multi_aff_set_aff(ma, i, aff);
387 }
388
389 pma = isl_pw_multi_aff_alloc(set, ma);
390 data->tiled = isl_pw_multi_aff_union_add(data->tiled, pma);
391
392 return isl_stat_ok;
393}

518/* Tile the given band using the specified tile sizes.
519 * The given band is modified to refer to the tile loops and
520 * a child band is created to refer to the point loops.
521 * The children of this point loop band are the children
522 * of the original band.
523 *
524 * If the scale tile loops option is set, then the tile loops
525 * are scaled by the tile sizes. If the shift point loops option is set,
526 * then the point loops are shifted to start at zero.
527 * In particular, these options affect the tile and point loop schedules
528 * as follows
529 *
530 * scale shift original tile point
531 *
532 * 0 0 i floor(i/s) i
533 * 1 0 i s * floor(i/s) i
534 * 0 1 i floor(i/s) i - s * floor(i/s)
535 * 1 1 i s * floor(i/s) i - s * floor(i/s)
536 */

538{

542 isl_union_pw_multi_aff *sched = NULL((void*)0), *child_sched = NULL((void*)0);
543 isl_space *space;
544 isl_multi_val *mv_sizes;
545
546 if (!band || !sizes)
547 goto error;
548
549 ctx = isl_vec_get_ctx(sizes);
550 child = isl_band_dup(band);
551 list = isl_band_list_alloc(ctx, 1);
552 list = isl_band_list_add(list, child);
553 if (!list)

555

559 sched = isl_union_pw_multi_aff_tile(sched, mv_sizes);
